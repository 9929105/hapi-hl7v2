44  /**
45   * A Runnable that accepts connections on a ServerSocket and adds them to a
46   * {@link BlockingQueue}, so that they can be handled asynchronously. After
47   * stop() is called, the ServerSocket is closed.
48   */
49  class AcceptorThread extends Service {
50  
51  	/**
52  	 * @deprecated See {@link StandardSocketFactory#DEFAULT_ACCEPTED_SOCKET_TIMEOUT}
53  	 */
54  	@Deprecated
55  	static final int TIMEOUT = 500;
56  	
57  	private static final Logger log = LoggerFactory
58  			.getLogger(AcceptorThread.class);
59  	private int port;
60  	private boolean tls = false;
61  	private ServerSocket ss;
62  	private final BlockingQueue<AcceptedSocket> queue;
63  	private final SocketFactory socketFactory;
64  
65  	public AcceptorThread(ServerSocket serverSocket, int port, ExecutorService service,
66  			BlockingQueue<AcceptedSocket> queue) throws IOException,
67  			SocketException {
68  		this(port, false, service, queue);
69  		this.ss = serverSocket;
70  	}
71  
72  	public AcceptorThread(int port, ExecutorService service,
73  			BlockingQueue<AcceptedSocket> queue) throws IOException,
74  			SocketException {
75  		this(port, false, service, queue);
76  	}
77  
78  	public AcceptorThread(int port, boolean tls, ExecutorService service,
79  			BlockingQueue<AcceptedSocket> queue) throws IOException,
80  			SocketException {
81  		this(port, tls, service, queue, null);
82  	}
83  
84  	public AcceptorThread(int port, boolean tls, ExecutorService service, BlockingQueue<AcceptedSocket> queue, SocketFactory socketFactory) {
85  		super("Socket Acceptor", service);
86  		this.port = port;
87  		this.queue = queue;
88  		this.tls = tls;
89  		if (socketFactory == null) {
90  			socketFactory = new StandardSocketFactory();
91  		}
92  		this.socketFactory = socketFactory;
93  	}
94  
95  	@Override
96  	protected void afterStartup() {
97  		super.afterStartup();
98  		try {
99  			if (this.tls) {
100 				ss = socketFactory.createTlsServerSocket();
101 			} else {
102 				ss = socketFactory.createServerSocket();
103 			}
104 			ss.bind(new InetSocketAddress(port));
105 			ss.setSoTimeout(500);
106 		} catch (IOException e) {
107 			final String message = String.format("Unable to create ServerSocket on port %d", port);
108 			throw new RuntimeException(message, e);
109 		}
110 	}
111 
112 	@Override
113 	protected void handle() {
114 		try {
115 			Socket s = ss.accept();
116 			socketFactory.configureNewAcceptedSocket(s);
117 			if (!queue.offer(new AcceptedSocket(s))) {
118 				log.error("Denied enqueuing server-side socket {}", s);
119 				s.close();
120 			} else
121 				log.debug("Enqueued server-side socket {}", s);
122 		} catch (SocketTimeoutException e) { /* OK - just timed out */
123 			log.trace("No connection established while waiting");
124 		} catch (IOException e) {
125 			log.error("Error while accepting connections", e);
126 		}
127 	}
128 
129 	@Override
130 	protected void afterTermination() {
131 		super.afterTermination();
132 		try {
133 			if (ss != null && !ss.isClosed())
134 				ss.close();
135 		} catch (IOException e) {
136 			log.warn("Error during stopping the thread", e);
137 		}
138 	}
139 
140 	class AcceptedSocket {
141 		Socket socket;
142 		AcceptorThread origin;
143 
144 		public AcceptedSocket(Socket socket) {
145 			if (socket == null)
146 				throw new IllegalArgumentException("Socket must not be null");
147 			this.socket = socket;
148 			this.origin = AcceptorThread.this;
149 		}
150 
151 	}
