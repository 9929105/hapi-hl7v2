40  public class GeneratedConformanceMessage extends GeneratedConformanceContainer {
41  
42     /** Default constructor */
43     public GeneratedConformanceMessage() {
44     }
45  
46     /** Adds the member variable to the ConformanceMessage class and
47      * adds a line to the constructor to instantiate that member variable
48      * @param a String representing the hl7 message type 
49      *        example "ca.uhn.hl7v2.model.ADT_A01"
50      */
51     public void addHAPIMessage(String msgType) {
52  
53        // add member variable
54        this.addMemberVariable("private " + msgType + " hapiMessage;");
55  
56        // add instantiator to the constructor
57        GeneratedMethod constructor = this.getConstructor();
58        constructor.addToBody("hapiMessage = new " + msgType + "();");
59  
60        // add getter for hapi message
61        GeneratedMethod getHAPIMessage = new GeneratedMethod();
62        getHAPIMessage.setName("getHAPIMessage");
63        getHAPIMessage.setVisibility("public");
64        getHAPIMessage.setReturnType(msgType);
65        getHAPIMessage.addToBody("return hapiMessage;");
66        getHAPIMessage.addToComments("This method returns the HAPI message object.");
67        getHAPIMessage.addToComments("@return A <code>" + msgType + "</code> object representing the type of message");
68        this.addMethod(getHAPIMessage);
69     }
