47  /**
48   * Create HAPI segment model classes from XML Schema files. Download the files from
49   * http://www.hl7.org/documentcenter/private/standards/V26/HL7-xml_v2.6_annotated.zip and
50   * http://www.hl7.org/documentcenter/public/standards/V2/Sun_HL7v2xsd.zip
51   * and unpack into src/main/resources/hl7v2xsd.
52   * <p/>
53   * This is an attempt to remove the need to have the HL7 database around because the
54   * schema files can be downloaded by any who has access to the HL7 standards.
55   */
56  public class XsdSegmentGenerator {

58      private static final Logger LOG = LoggerFactory.getLogger(XsdSegmentGenerator.class);
59      private static final String[] EXCLUDE_SEGMENTS = {"anyHL7Segment", "anyZSegment"};

61      public static final String URN_HL7_ORG_V2XML = "urn:hl7-org:v2xml";

64      private String targetDirectory;

66      public XsdSegmentGenerator(String dir, String templatePackage) throws IOException {
67          File f = new File(dir);
68          if (!f.isDirectory())
69              throw new IOException("Can't create file in " + dir + " - it is not a directory.");
70          this.targetDirectory = dir;
71          this.templatePackage = templatePackage.replace(".", "/");
72      }

74      public void parse(Version version) throws Exception {
75          XSOMParser parser = new XSOMParser();
76          String dir = String.format("/hl7v2xsd/%s/segments.xsd", version.getVersion());
77          URL url = getClass().getResource(dir);
78          parser.parse(url);
79          XSSchemaSet result = parser.getResult();
80          Iterator<XSSchema> iter = result.iterateSchema();
81          while (iter.hasNext()) {
82              XSSchema schema = iter.next();
83              if (URN_HL7_ORG_V2XML.equals(schema.getTargetNamespace())) {
84                  parseSegments(schema, version);
85              }
86          }
87      }

90      //<xsd:sequence>
91      //<xsd:any namespace="##other" processContents="lax" minOccurs="0"/>
92      //</xsd:sequence>
93      //</xsd:complexType>
94      //<xsd:element name="PD1" type="PD1.CONTENT"/>

96      private void parseSegments(XSSchema schema, Version version) throws Exception {
97  
98          Template template = VelocityFactory.getClasspathTemplateInstance(templatePackage + "/segment.vsm");
99          String basePackageName = DefaultModelClassFactory.getVersionPackageName(version.getVersion());
100         String[] datatypePackages = { basePackageName + "datatype" };
101 
102         Iterator<XSElementDecl> segmentDecls = schema.iterateElementDecls();
103         while (segmentDecls.hasNext()) {
104             XSElementDecl segmentDecl = segmentDecls.next();
105             String segmentName = segmentDecl.getName();
106             if (isRealSegment(segmentName)) {
107 
108                 List<SegmentElement> segmentsElements = new ArrayList<SegmentElement>();
109                 XSComplexType complexType = segmentDecl.getType().asComplexType();
110                 // Find and iterate over the fields of the segment
111                 XSParticle[] children = complexType
112                         .getContentType()
113                         .asParticle()
114                         .getTerm()
115                         .asModelGroup()
116                         .getChildren();
117                 LOG.debug("Creating segment {}, having {} children", segmentName, children.length);
118                 for (int i = 0; i < children.length; i++) {
119                     // Navigate to the attributes
120                     if (!children[i].getTerm().isWildcard()) {
121                         XSAttGroupDecl attrGroup = children[i]
122                                 .getTerm()
123                                 .asElementDecl()
124                                 .getType()
125                                 .asComplexType()
126                                 .getAttGroups().iterator().next();
127                         LOG.debug("Field {}", attrGroup.getName());
128                         String fieldType = attrGroup.getAttributeUse("", "Type").getFixedValue().toString();
129                         String fieldDescription = attrGroup.getAttributeUse("", "LongName").getFixedValue().toString();
130                         XSAttributeUse fieldLength = attrGroup.getAttributeUse("", "maxLength");
131                         int maxLength = 0;
132                         if (fieldLength != null)
133                             maxLength = Integer.parseInt(fieldLength.getFixedValue().toString());
134                         XSAttributeUse fieldTable = attrGroup.getAttributeUse("", "Table");
135                         int table = 0;
136                         if (fieldTable != null)
137                             table = Integer.parseInt(fieldTable.getFixedValue().toString().substring(3));
138                         SegmentElement se = new SegmentElement(segmentName, version.getVersion(), i);
139                         se.field = i;
140                         if (children[i].getMaxOccurs().intValue() == XSParticle.UNBOUNDED) {
141                             se.rep = "Y";
142                             se.repetitions = 0;
143                         } else {
144                             se.rep = "N";
145                             se.repetitions = 1;
146                         }
147                         se.opt = children[i].getMinOccurs().intValue() > 0 ? "R" : "O";
148                         se.desc = fieldDescription;
149                         se.table = table;
150                         se.type = fieldType;
151                         fixType(i + 1, version, se);
152                         se.length = maxLength;
153                         segmentsElements.add(se);
154                     }
155                 }
156                 // need to add better segement description here, but nothing in XSD!
157                 writeSegment(template, basePackageName, datatypePackages,
158                         segmentName, segmentName, version, segmentsElements);
159             }
160 
161         }
162 
163     }

165     private void writeSegment(Template template, String basePackageName, String[] datatypePackages,
166                               String segmentName, String segmentDescription, Version version,
167                               List<SegmentElement> segmentsElements) throws Exception {
168         String source = make(template, basePackageName, datatypePackages,
169                 segmentName, segmentDescription, version.getVersion(), segmentsElements);
170         if (source != null) {
171             writeFile(source, segmentName, version);
172         }
173     }

175     /**
176      * This method can be used to circumvent errors in the XSDs without the need to
177      * modify the XDS.
178      *
179      * @param index
180      * @param version
181      * @param se
182      */
183     private static void fixType(int index, Version version, SegmentElement se) {
184         // Null/withdrawn
185         if (se.type.equals("NUL") || se.type.equals("-")) {
186             se.type = "NULLDT";
187 
188         // 3454369
189         } else if (version == Version.V23 && se.type.equals("MRG") && index == 7) {
190             se.type = "XPN";
191 
192         // https://sourceforge.net/p/hl7api/bugs/95/
193         } else if (version == Version.V23 && se.type.equals("ORC") && index == 14) {
194             se.type = "XTN";
195 
196         // 2864817
197         } else if (version == Version.V23 && se.type.equals("PID") && index == 5) {
198             se.rep = "Y";
199             se.repetitions = -1;
200         }
201     }

203     private boolean isRealSegment(String dataTypeName) {
204         return Arrays.binarySearch(EXCLUDE_SEGMENTS, dataTypeName) < 0 &&
205                 dataTypeName.indexOf('.') < 0;
206     }

208     private void writeFile(String source, String segmentName, Version version) throws IOException {
209         // TODO must be more robust
210         String dirName = String.format("%s/ca/uhn/hl7v2/model/%s/segment/", targetDirectory, version.getPackageVersion());
211         File dir = new File(dirName);
212         if (!dir.exists()) {
213             dir.mkdirs();
214         }
215         String targetFile = String.format("%s/%s.java", dirName, segmentName);
216 
217         BufferedWriter writer = null;
218         try {
219             writer = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(targetFile, false), "UTF-8"));
220             writer.write(source);
221             writer.flush();
222         } finally {
223             if (writer != null) writer.close();
224         }
225     }

227     private String make(Template template, String normalBasePackageName, String[] datatypePackages,
228                         String segmentName, String description, String version, List<SegmentElement> elements) throws Exception {
229         StringWriter out = new StringWriter();
230         Context ctx = new VelocityContext();
231         ctx.put("segmentName", segmentName);
232         ctx.put("typeDescription", description);
233         ctx.put("basePackageName", normalBasePackageName);
234         ctx.put("elements", elements);
235         ctx.put("datatypePackages", datatypePackages);
236         ctx.put("hl7VersionInQuotes", '"' + version + '"');
237 
238         template.merge(ctx, out);
239         return out.toString();
240     }
