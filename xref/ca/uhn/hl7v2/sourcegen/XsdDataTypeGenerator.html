45  /**
46   * Create HAPI datatype model classes from XML Schema files. Download the files from
47   * http://www.hl7.org/documentcenter/private/standards/V26/HL7-xml_v2.6_annotated.zip and
48   * http://www.hl7.org/documentcenter/public/standards/V2/Sun_HL7v2xsd.zip
49   * and unpack into src/main/resources/hl7v2xsd.
50   * <p>
51   * This is an attempt to remove the need to have the HL7 database around because the
52   * schema files can be downloaded by any who has access to the HL7 standards.
53   */
54  public class XsdDataTypeGenerator {
55  
56      private static final String[] PRIMITIVES = {"FT", "GTS", "NM", "SI", "ST", "TN",  "TX"};
57      private static final String[] EXCLUDE_COMPOSITES = {"TX_CHALLENGE", "escapeType", "varies"};
58  
59      public static final String URN_HL7_ORG_V2XML = "urn:hl7-org:v2xml";
60  
61      private String templatePackage;
62      private String targetDirectory;
63  
64      public XsdDataTypeGenerator(String dir, String templatePackage) throws IOException {
65          File f = new File(dir);
66          if (!f.isDirectory())
67              throw new IOException("Can't create file in " +
68                      dir.toString() + " - it is not a directory.");
69          this.targetDirectory = dir;
70          this.templatePackage = templatePackage.replace(".", "/");
71      }
72  
73      public void parse(Version version) throws Exception {
74          XSOMParser parser = new XSOMParser();
75          String dir = String.format("/hl7v2xsd/%s/datatypes.xsd", version.getVersion());
76          URL url = getClass().getResource(dir);
77          parser.parse(url);
78          XSSchemaSet result = parser.getResult();
79          Iterator<XSSchema> iter = result.iterateSchema();
80          while (iter.hasNext()) {
81              XSSchema schema = iter.next();
82              if (URN_HL7_ORG_V2XML.equals(schema.getTargetNamespace())) {
83                  parsePrimitives(schema, version);
84                  parseComposites(schema, version);
85              }
86          }
87      }
88  
89      private void parsePrimitives(XSSchema schema, Version version) throws Exception {
90          List<DatatypeDef> primitiveTypes = new ArrayList<DatatypeDef>();
91          Iterator<XSType> types = schema.iterateTypes();
92  
93          while (types.hasNext()) {
94              XSType type = types.next();
95              String dataTypeName = type.getName();
96              if (Arrays.binarySearch(PRIMITIVES, dataTypeName) >= 0)
97                  primitiveTypes.add(new DatatypeDef(dataTypeName, dataTypeName));
98          }
99  
100         Template template = VelocityFactory.getClasspathTemplateInstance(templatePackage + "/datatype_primitive.vsm");
101         String basePackageName = DefaultModelClassFactory.getVersionPackageName(version.getVersion());
102         String normalBasePackageName = DefaultModelClassFactory.getVersionPackageName(version.getVersion());
103 
104         for (DatatypeDef def : primitiveTypes) {
105             String source = make(template, basePackageName, normalBasePackageName, def, version.getVersion());
106             if (source != null) {
107                 writeFile(source, def, version);
108             }
109         }
110     }
111 
112     private void parseComposites(XSSchema schema, Version version) throws Exception {
113         List<DatatypeDef> compositeTypes = new ArrayList<DatatypeDef>();
114         Iterator<XSComplexType> types = schema.iterateComplexTypes();
115         while (types.hasNext()) {
116             XSComplexType complexType = types.next();
117             String dataTypeName = complexType.getName();
118 
119             // Omit CE_X types
120             if (dataTypeName.startsWith("CE_")) dataTypeName = "CE";
121 
122             if (isRealComposite(dataTypeName)) {
123                 DatatypeDef compositeType = new DatatypeDef(dataTypeName, dataTypeName);
124                 // Extract list of components from the composite type
125                 XSParticle[] children = complexType
126                         .getContentType()
127                         .asParticle()
128                         .getTerm()
129                         .asModelGroup()
130                         .getChildren();
131                 // Iterate over all components
132                 for (int i = 0; i < children.length; i++) {
133                     XSAttGroupDecl attrGroup = children[i]
134                             .getTerm()
135                             .asElementDecl()
136                             .getType()
137                             .asComplexType()
138                             .getAttGroups().iterator().next();
139                     String componentType = attrGroup.getAttributeUse("", "Type").getFixedValue().toString();
140                     String componentDescription = attrGroup.getAttributeUse("", "LongName").getFixedValue().toString();
141                     XSAttributeUse componentTable = attrGroup.getAttributeUse("", "Table");
142                     int table = 0;
143                     if (componentTable != null)
144                         table = Integer.parseInt(componentTable.getFixedValue().toString().substring(3));
145 
146                     compositeType.addSubcomponentDef(
147                             new DatatypeComponentDef(
148                                     dataTypeName,
149                                     i,
150                                     fixTypeName(dataTypeName, componentType),
151                                     componentDescription,
152                                     table));
153                 }
154                 compositeTypes.add(compositeType);
155             }
156 
157         }
158 
159         Template template = VelocityFactory.getClasspathTemplateInstance(templatePackage + "/datatype_composite.vsm");
160         String basePackageName = DefaultModelClassFactory.getVersionPackageName(version.getVersion());
161         String normalBasePackageName = DefaultModelClassFactory.getVersionPackageName(version.getVersion());
162 
163         for (DatatypeDef def : compositeTypes) {
164             String source = make(template, basePackageName, normalBasePackageName, def, version.getVersion());
165             if (source != null) {
166                 writeFile(source, def, version);
167             }
168         }
169     }

171     // Really need all of this?
172     private static String fixTypeName(String parentName, String dataTypeName) {
173         if (dataTypeName.equals("ST") && parentName.equals("TS")) return "TSComponentOne";
174         //convert to varies to Varies
175         if (dataTypeName.equals("varies")) return "Varies";
176 
177         // Null/withdrawn
178         if (dataTypeName.equals("NUL")) return "NULLDT";
179         if (dataTypeName.equals("-")) return "NULLDT";
180 
181         return dataTypeName;
182     }
183 
184     private boolean isRealComposite(String dataTypeName) {
185         return Arrays.binarySearch(PRIMITIVES, dataTypeName) < 0 &&
186                 Arrays.binarySearch(EXCLUDE_COMPOSITES, dataTypeName) < 0 &&
187                 dataTypeName.indexOf('.') < 0;
188     }

190     private void writeFile(String source, DatatypeDef def, Version version) throws IOException {
191         // TODO must be more robust
192         String dirName = String.format("%s/ca/uhn/hl7v2/model/%s/datatype/", targetDirectory, version.getPackageVersion());
193         File dir = new File(dirName);
194         if (!dir.exists()) {
195             dir.mkdirs();
196         }
197         String targetFile = String.format("%s/%s.java", dirName, def.getType());
198 
199         BufferedWriter writer = null;
200         try {
201             writer = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(targetFile, false), "UTF-8"));
202             writer.write(source);
203             writer.flush();
204         } finally {
205             if (writer != null) writer.close();
206         }
207     }

209     private String make(Template template, String basePackageName, String normalBasePackageName,
210                         DatatypeDef def, String version) throws Exception {
211         StringWriter out = new StringWriter();
212         Context ctx = new VelocityContext();
213         ctx.put("datatype", def);
214         ctx.put("datatypeName", def.getType());
215         ctx.put("version", version);
216         ctx.put("basePackageName", basePackageName);
217         ctx.put("normalBasePackageName", normalBasePackageName);
218         ctx.put("components", def.getSubComponentDefs());
219         ctx.put("desc", def.getName());
220 
221         template.merge(ctx, out);
222         return out.toString();
223     }
