70      /**
71       * <p>Creates skeletal source code (without correct data structure but no business
72       * logic) for all data types found in the normative database.  For versions > 2.2, Primitive data types
73       * are not generated, because they are coded manually (as of HAPI 0.3).  
74      * @param theTemplatePackage 
75       */
76      public static void makeAll(String baseDirectory, String version, String theTemplatePackage, String theFileExt) throws IOException, SQLException, HL7Exception {
77          //make base directory
78          if (!(baseDirectory.endsWith("\\") || baseDirectory.endsWith("/"))) { 
79              baseDirectory = baseDirectory + "/";
80          }
81          File targetDir = SourceGenerator.makeDirectory(baseDirectory + DefaultModelClassFactory.getVersionPackagePath(version) + "datatype"); 
82          
83          //get list of data types
84          
85          ArrayList<String> types = getDataTypes(version);
86          
87          System.out.println("Generating " + types.size() + " datatypes for version " + version);
88          if (types.size() == 0) { 
89              log.warn("No version {} data types found in database {}", 
90              		version, System.getProperty("ca.on.uhn.hl7.database.url"));
91          }
92                  
93          for (int i = 0; i < types.size(); i++) {
94              try {
95                  String basePackage = DefaultModelClassFactory.getVersionPackageName(version);
96                  String dataType = (String)types.get(i);
97                  
98  				String hapiTestGenType = System.getProperty("hapi.test.gentype");
99  				if (hapiTestGenType != null && !hapiTestGenType.contains(dataType)) {
100 					continue;
101 				}
102                 
103 				make(targetDir, dataType, version, basePackage, theTemplatePackage, theFileExt);
104             } catch (DataTypeException dte) {
105                 log.warn("{} - {}", dte.getClass().getName(), dte.getMessage());
106             } catch (Exception e) {
107                 log.error("Error creating source code for all data types", e);
108             } 
109         }
110     }

112     public static ArrayList<String> getDataTypes(String version) throws SQLException {
113         ArrayList<String> types = new ArrayList<String>();
114         NormativeDatabase normativeDatabase = NormativeDatabase.getInstance();
115         Connection conn = normativeDatabase.getConnection();
116         Statement stmt = conn.createStatement();
117         //get normal data types ... 
118         ResultSet rs = stmt.executeQuery("select data_type_code from HL7DataTypes, HL7Versions where HL7Versions.version_id = HL7DataTypes.version_id and HL7Versions.hl7_version = '" + version + "'");
119         while (rs.next()) {
120             types.add(rs.getString(1));
121         }
122         
123         
124         //get CF, CK, CM, CN, CQ sub-types ... 
125  
126        rs = stmt.executeQuery("select data_structure from HL7DataStructures, HL7Versions where (" + 
127             "data_type_code  = 'CF' or " + 
128             "data_type_code  = 'CK' or " + 
129             "data_type_code  = 'CM' or " + 
130             "data_type_code  = 'CN' or " + 
131             "data_type_code  = 'CQ') and " +
132 	    "HL7Versions.version_id = HL7DataStructures.version_id and  HL7Versions.hl7_version = '" + version + "'");
133         while (rs.next()) {
134             String string = rs.getString(1);
135             if (string.equals("-")) {
136                 continue;
137             }
138             
139             types.add(string);
140         }
141         
142         stmt.close();
143         normativeDatabase.returnConnection(conn);
144         return types;
145     }

147     /**
148      * Creates source code for a single data type in the HL7 normative
149      * database. 
150      * @param targetDirectory the directory into which the file will be written
151      * @param datatype the name (e.g. ST, ID, etc.) of the data type to be created
152      * @param version the HL7 version of the intended data type
153      * @param theFileExt 
154      */
155     public static void make(File targetDirectory, String dataType, String version, String basePackage, String theTemplatePackage, String theFileExt) throws Exception {
156         //make sure that targetDirectory is a directory ... 
157         if (!targetDirectory.isDirectory()) throw new IOException("Can't create file in " + 
158             targetDirectory.toString() + " - it is not a directory.");
159                 
160         //get any components for this data type
161         NormativeDatabase normativeDatabase = NormativeDatabase.getInstance();
162         Connection conn = normativeDatabase.getConnection();
163         Statement stmt = conn.createStatement();
164         StringBuffer sql = new StringBuffer();
165         //this query is adapted from the XML SIG informative document
166         sql.append("SELECT HL7DataStructures.data_structure, HL7DataStructureComponents.seq_no, HL7DataStructures.description, HL7DataStructureComponents.table_id,  ");
167         sql.append("HL7Components.description, HL7Components.table_id, HL7Components.data_type_code, HL7Components.data_structure ");
168         sql.append("FROM HL7Versions LEFT JOIN (HL7DataStructures LEFT JOIN (HL7DataStructureComponents LEFT JOIN HL7Components ");
169         sql.append("ON HL7DataStructureComponents.comp_no = HL7Components.comp_no AND ");
170         sql.append("HL7DataStructureComponents.version_id = HL7Components.version_id) ");
171         sql.append("ON HL7DataStructures.version_id = HL7DataStructureComponents.version_id ");
172         sql.append("AND HL7DataStructures.data_structure = HL7DataStructureComponents.data_structure) ");
173         sql.append("ON HL7DataStructures.version_id = HL7Versions.version_id ");
174         sql.append("WHERE HL7DataStructures.data_structure = '");
175         sql.append(dataType);
176         sql.append("' AND HL7Versions.hl7_version = '");
177         sql.append(version);
178         sql.append("' ORDER BY HL7DataStructureComponents.seq_no");
179         //System.out.println(sql.toString());  //for debugging
180         ResultSet rs = stmt.executeQuery(sql.toString());
181         
182         ArrayList<String> dataTypes = new ArrayList<String>(20);
183         ArrayList<String> descriptions = new ArrayList<String>(20);
184         ArrayList<Integer> tables = new ArrayList<Integer>(20);
185         String description = null;
186         while (rs.next()) {
187             if (description == null) description = rs.getString(3);
188 
189             String de = rs.getString(5);
190             String dt = rs.getString(8);
191             int ta = rs.getInt(4);
192             //trim all CE_x to CE
193             if (dt != null) if (dt.startsWith("CE")) dt = "CE";
194             //System.out.println("Component: " + de + "  Data Type: " + dt);  //for debugging
195 
196             // Prior to HL7 2.5, the first component of a TS was
197             // an undefined component HAPI knows as TSComponentOne, but the
198             // database knows it as an ST
199             if (dataType.equals("TS") && "ST".equals(dt) && dataTypes.isEmpty()) {
200                 dt = "TSComponentOne";
201             }
202 
203             dataTypes.add(dt);
204             descriptions.add(de);
205             tables.add(new Integer(ta));
206         }
207         stmt.close();
208         normativeDatabase.returnConnection(conn);
209         
210         //if there is only one component make a Primitive, otherwise make a Composite
211         String source = null;
212         if (dataTypes.size() == 1) {
213             if (ourMakeAll || dataType.equals("FT") || dataType.equals("ST") || dataType.equals("TX") 
214                     || dataType.equals("NM") || dataType.equals("SI") || dataType.equals("TN")
215                     || dataType.equals("GTS")) { 
216                 source = makePrimitive(new DatatypeDef(dataType, description), version, basePackage, theTemplatePackage);                
217             } else {
218                 source = null; //note: IS, ID, DT, DTM, and TM are coded manually
219             }
220         } else if (dataTypes.size() > 1) {
221             int numComponents = dataTypes.size();
222             //copy data into arrays ... 
223             String[] type = new String[numComponents];
224             String[] desc = new String[numComponents];
225             int[] table = new int[numComponents];
226             DatatypeComponentDef[] componentDefs = new DatatypeComponentDef[numComponents];
227             Set<String> names = new HashSet<String>();
228             for (int i = 0; i < numComponents; i++) {
229                 type[i] = (String)dataTypes.get(i);
230                 String componentName = (String)descriptions.get(i);
231                 
232                 if (names.contains(componentName)) {
233                	 for (int j = 2; ; j++) {
234                		 if (!names.contains(componentName + j)) {
235                			 componentName = componentName + j;
236                			 break;
237                		 }
238                	 }
239                 }
240                 names.add(componentName);
241                 
242                 desc[i] = componentName;
243                 table[i] = ((Integer) tables.get(i)).intValue();
244                 
245                 String typeName = (String)dataTypes.get(i);
246                 typeName = SourceGenerator.getAlternateType(typeName, version);
247                 
248                 componentDefs[i] = new DatatypeComponentDef(dataType, i, typeName, componentName, ((Integer) tables.get(i)).intValue());
249             }
250             
251             source = makeComposite(dataType, description, componentDefs, type, desc, table, version, basePackage, theTemplatePackage);
252         } else {
253             if (dataType.equals("var")) {
254                 return; // Varies isn't actually a type
255             }
256             
257             //no components?  
258             throw new DataTypeException("The data type " + dataType + " could not be found");
259         }
260         
261         //System.out.println(source);
262         
263         //write to file ... 
264         if (source != null) {
265             String targetFile = targetDirectory.toString() + "/" + dataType + "." + theFileExt;
266             BufferedWriter writer = null;
267             try {
268 	            writer = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(targetFile, false), SourceGenerator.ENCODING));
269 	            writer.write(source);
270 	            writer.flush();
271             } finally {
272             	if (writer != null) writer.close();
273             }
274         }
275     }

278     /**
279      * Returns a String containing the complete source code for a Primitive HL7 data
280      * type.  Note: this method is no longer used, as all Primitives are now coded manually.  
281      */
282     private static String makePrimitive(DatatypeDef theDatatype, String version, String basePackage, String theTemplatePackage) throws Exception {
283 
284             StringWriter out = new StringWriter();
285 
286             theTemplatePackage = theTemplatePackage.replace(".", "/");
287             Template template = VelocityFactory.getClasspathTemplateInstance(theTemplatePackage + "/datatype_primitive.vsm");
288             Context ctx = new VelocityContext();
289             ctx.put("datatype", theDatatype);
290             ctx.put("version", version);
291             ctx.put("basePackageName", basePackage);
292             ctx.put("normalBasePackageName", DefaultModelClassFactory.getVersionPackageName(version));
293             
294             template.merge(ctx, out);
295             return out.toString();
296             
297     }

299     /**
300      * Returns a String containing source code for a Composite data type. The 
301      * dataTypes array contains the data type names (e.g. ST) of each component. 
302      * The descriptions array contains the corresponding descriptions (e.g. string).
303      */
304     private static String makeComposite(String dataType, String description, DatatypeComponentDef[] componentDefs, String[] dataTypes, 
305             String[] descriptions, int[] tables, String version, String basePackage, String theTemplatePackage) throws Exception {
306         
307             StringWriter out = new StringWriter();
308 
309             for (int i = 0; i < componentDefs.length; i++) {
310             	if (componentDefs[i].getType().equals(dataType)) {
311                   log.warn("Datatype {} has a component child also of type {}! Can not recurse like this", dataType, dataType);
312                   componentDefs[i].setType("ST");	
313             	}
314             }
315             
316             for (int i = 0; i < dataTypes.length; i++) {
317                if (dataTypes[i].equals(dataType)) {
318                   log.warn("Datatype {} has a child also of type {}! Can not recurse like this", dataType, dataType);
319                   dataTypes[i] = "ST";
320                }
321             }
322             
323             theTemplatePackage = theTemplatePackage.replace(".", "/");
324             Template template = VelocityFactory.getClasspathTemplateInstance(theTemplatePackage + "/datatype_composite.vsm");
325             Context ctx = new VelocityContext();
326             ctx.put("datatypeName", dataType);
327             ctx.put("version", version);
328             ctx.put("basePackageName", basePackage);
329             ctx.put("components", Arrays.asList(componentDefs));
330             ctx.put("desc", description);
331             
332             template.merge(ctx, out);
333             return out.toString();
334             
335     }

358     public static void writeDatatype(String theFileName, String theVersion, DatatypeDef theFieldDef, String theBasePackage, String theTemplatePackage) throws Exception {
359         
360         String text;
361         if (theFieldDef.getSubComponentDefs().isEmpty()) {
362             text = makePrimitive(theFieldDef, theVersion, theBasePackage, theTemplatePackage);
363         } else {
364             text = makeComposite(theFieldDef.getType(), theFieldDef.getName(), theFieldDef.getSubComponentDefs().toArray(new DatatypeComponentDef[0]), null, null, null, theVersion, theBasePackage, theTemplatePackage);
365         }
366 
367         BufferedWriter writer = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(theFileName, false), SourceGenerator.ENCODING));
368         writer.write(text);
369         writer.flush();
370         writer.close();
371         
372     }
