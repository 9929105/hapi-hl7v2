029package ca.uhn.hl7v2.llp;
030
031import java.io.*;
032
033/**
034 * An HL7Writer implements writing to an output stream using a particular lower 
035 * layer protocol.
036 *
037 * @author Bryan Tripp
038 */
039public interface HL7Writer {
040
041    /** 
042     * Sends a complete message to the underlying output stream using the  
043     * lower layer protocol implemented by this HL7Writer.  
044     */
045    public void writeMessage(String message) throws LLPException, IOException;
046    
047    /**
048     * Sets the OutputStream to which to write messages.  The OutputStream must be set 
049     * before any calls to <code>()</code>.
050     */
051    public void setOutputStream(OutputStream out) throws IOException;
052
053    /** 
054     * Closes the underlying stream and any other resources that should be closed. 
055     */
056    public void close() throws java.io.IOException;
057    
058}
