027package ca.uhn.hl7v2.llp;
028
029import java.io.IOException;
030import java.io.InputStream;
031import java.nio.charset.Charset;
032
033public class MinLLPReader extends Hl7DecoderReader<MllpDecoder> {
034
035        /**
036         * @deprecated Use {@link MllpConstants#CHARSET_KEY} instead
037         */
038        @Deprecated()
039    public static final String CHARSET_KEY = MllpConstants.CHARSET_KEY;
040
041    public MinLLPReader() throws IOException {
042    }
043
044    public MinLLPReader(InputStream in) throws IOException {
045        super(in);
046    }
047
048    public MinLLPReader(InputStream in, Charset charset) throws IOException {
049        super(in, charset);
050    }
051
052    @Override
053    protected MllpDecoder initDecoder() {
054        return new MllpDecoder(getCharset());
055    }
056
057
058}
