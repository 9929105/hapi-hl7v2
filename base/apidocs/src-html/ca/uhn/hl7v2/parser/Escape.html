027package ca.uhn.hl7v2.parser;
028
029/**
030 * Deprecated static implementation; simply delegates to an instance
031 * of {@link DefaultEscaping}.
032 *
033 * @author Christian Ohr
034 * @deprecated use {@link DefaultEscaping} instance
035 */
036public final class Escape {
037
038    private static final Escaping DEFAULT = new DefaultEscaping();
039
040    private Escape() {}
041
042    /**
043     * @param text string to be escaped
044     * @param encChars encoding characters to be used
045     * @return the escaped string
046     */
047    public static String escape(String text, EncodingCharacters encChars) {
048        return DEFAULT.escape(text, encChars);
049    }
050
051    /**
052     * @param text string to be unescaped
053     * @param encChars encoding characters to be used
054     * @return the unescaped string
055     */
056    public static String unescape(String text, EncodingCharacters encChars) {
057        return DEFAULT.unescape(text, encChars);
058    }
059
060}
