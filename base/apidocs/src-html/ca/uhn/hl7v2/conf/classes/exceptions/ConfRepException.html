040@SuppressWarnings("serial")
041public class ConfRepException extends ConformanceException {
042
043   /** Constructor for ConfRepException
044    * @param val the error message to throw
045    */
046   public ConfRepException( int maxReps, int desiredRep ) {
047      super("Illegal repetition requested. Note that repetition indexes start at 1. Desired repetition: " + desiredRep + ", Maximum allowable: " + maxReps);
048   }
049
050}
