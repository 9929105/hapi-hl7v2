026package ca.uhn.hl7v2.validation.impl;
027
028import java.util.StringTokenizer;
029
030import ca.uhn.hl7v2.validation.MessageRule;
031
032/**
033 * A <code>RuleBinding</code> that applies to <code>MessageRule</code>s. 
034 * <p>
035 * In general there is no need to use this class directly. Instead use the
036 * Validation builder classes to generate rules and rule bindings.
037 *  
038 * @author Bryan Tripp
039 */
040@SuppressWarnings("serial")
041public class MessageRuleBinding extends RuleBinding<MessageRule> {
042
043    private String myMessageType;
044    private String myTriggerEvent;
045    private static final String delim = "^"; 
046    
047    /**
048     * @param theVersion see {@link RuleBinding#getVersion()}
049     * @param theMessageType a value valid for MSH-9-1 or * for all
050     * @param theTriggerEvent a value valid for MSH-9-2 or * for all
051     * @param theRule rule to be used for binding
052     */
053    public MessageRuleBinding(String theVersion, String theMessageType, String theTriggerEvent, MessageRule theRule) {
054        super(theVersion, theMessageType + "^" + theTriggerEvent, theRule);
055        myMessageType = theMessageType;
056        myTriggerEvent = theTriggerEvent;
057    }
058
059    /**
060     * @param theScope must be in the form type^event (ie message type and trigger event separated
061     *      by a caret)
062     */
063    public boolean appliesToScope(String theScope) {
064        
065        if (!theScope.matches(".*\\^.*")) {
066            throw new IllegalArgumentException("arg theScope must be of the form type^event");
067        }
068        
069        StringTokenizer tok = new StringTokenizer(theScope, delim, false);
070        String itemType = tok.nextToken();
071        String itemEvent = tok.nextToken();
072        
073        return applies(myMessageType, itemType) && applies(myTriggerEvent, itemEvent);
074    }
075
076}
